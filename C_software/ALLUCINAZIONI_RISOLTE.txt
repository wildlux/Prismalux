        out_c = "P: Il bit è l'unità minima." # Concordanza con A
        out_b = "ATTENZIONE: Esiste il Qubit, il Bit non è l'unica unità minima!" # Opposto trova una falla

        print(f"[1] Agente A: {out_a}")
        print(f"[2] Agente C (Gemello): {out_c}")
